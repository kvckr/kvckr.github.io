Query Impl(v1, v2)
    h1 = Depth(v1)
    h2 = Depth(v2)
    while h1 ≠ h2 do
        if h1 > h2 then
            v1 = Parent(v1)
            h1 = h1 - 1
        else
            v2 = Parent(v2)
            h2 = h2 - 1
    while v1 ≠ v2 do
        v1 = Parent(v1)
        v2 = Parent(v2)
    return v1

  end

        break
      else
        current = current.right
      end
    else
      if current.left == nil
        current.left = Node.new(new_value)

      else
